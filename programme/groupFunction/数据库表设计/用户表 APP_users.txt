create table APP_users(
U_ID int not null auto_increment  primary key COMMENT '主键、自增',
U_LoginID varchar(20) not null comment '登录账号',
U_PassWord Varchar(20)not null comment '密码',
U_SignaTure Varchar(150) comment '个性签名',
U_Sex int(1) comment '性别 0:女，1:男 ',
U_Birthday Datetime comment '生日',
U_Telephone Varchar(30) comment '电话',
U_Name Varchar(30)  comment '真实姓名',
U_Email Varchar(50)  comment '邮箱',
U_Intro Varchar(300)  comment '简介' ,
U_HeadPortrait Varchar(100)  comment '头像' ,
U_ShengXiao Char(2)  comment '生肖' ,
U_Age  Int    comment '年龄',
U_Constellation  Char(6)  comment '星座' ,
U_BloodType  Varchar(10)  comment '血型' ,
U_SchoolTag  Varchar(50)  comment '毕业学校' ,
U_Vocation  Varchar(30)  comment '职业' ,
U_NationID  Int  comment '国家ID'  ,
U_ProvinceID  Int  comment '省份ID'  ,
U_CityID  Int  comment '城市ID'  ,
U_FriendshipPolicyID  Int  comment '好友策略ID'  ,
U_UserStateID  Int  comment '用户状态ID'  ,
U_FriendPolicyQuestion  Varchar(30)  comment '好友策略问题' ,
U_FriendPolicyAnswer  Varchar(30)  comment '好友策略答案' ,
U_FriendPolicyPassword  Varchar(30)  comment '好友策略密码' 
);
